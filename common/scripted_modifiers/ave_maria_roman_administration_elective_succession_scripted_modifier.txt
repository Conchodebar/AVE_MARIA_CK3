elector_voting_pattern_opinion_roman_administration_elective_modifier = {
	
	## Junior Emperor
	modifier = { #Raw Opinion bonus (positive)
		desc = tooltip_roman_administration_junior_emperor_vote_opinion_positive
		NOT = {
			has_relation_rival = scope:candidate #Completely removed for rivals.
		}
		NOT = { this = scope:candidate } #Not on yourself.
		scope:candidate = {
			has_trait = junior_emperor
		}
		add = {
			value = 100
			if = { #Even more so if higher opinion.
				limit = {
					opinion = {
						target = scope:candidate
						value >= medium_positive_opinion
					}
				}
				add = 15 #30 total
			}
			if = { #Even more so if higher opinion.
				limit = {
					opinion = {
						target = scope:candidate
						value >= high_positive_opinion
					}
				}
				add = 20 #50 total
			}
			if = { #Even more so if higher opinion.
				limit = {
					opinion = {
						target = scope:candidate
						value >= very_high_positive_opinion
					}
				}
				add = 20 #70 total
			}
			if = { #Even more so if friend/lover.
				limit = {
					OR = {
						has_relation_friend = scope:candidate
						has_relation_lover = scope:candidate
					}
				}
				add = 30 #100 total
			}
			if = { #Further increased for Tanistry.
				limit = {
					scope:title = { has_title_law = gaelic_elective_succession_law }
				}
				multiply = 1.5
			}
		}
	}

	## legitimacy
	modifier = {
		desc = tooltip_roman_administration_legitimacy_vote_opinion_positive
		NOT = { this = scope:candidate }
		scope:candidate = {
			has_variable = roman_administration_legitimacy
			var:roman_administration_legitimacy > 0
		}
		add = {
			add = 1
			if = {
				limit = {
					scope:candidate = {
						has_variable = roman_administration_legitimacy
						var:roman_administration_legitimacy > 0
						var:roman_administration_legitimacy <= 50
					}					
				}
				add = 50
			}
			if = {
				limit = {
					scope:candidate = {
						has_variable = roman_administration_legitimacy
						var:roman_administration_legitimacy > 50
						var:roman_administration_legitimacy <= 60
					}					
				}
				add = 100
			}
			if = {
				limit = {
					scope:candidate = {
						has_variable = roman_administration_legitimacy
						var:roman_administration_legitimacy > 60
						var:roman_administration_legitimacy <= 70
					}					
				}
				add = 150
			}
			if = {
				limit = {
					scope:candidate = {
						has_variable = roman_administration_legitimacy
						var:roman_administration_legitimacy > 70
						var:roman_administration_legitimacy <= 80
					}					
				}
				add = 200
			}
			if = {
				limit = {
					scope:candidate = {
						has_variable = roman_administration_legitimacy
						var:roman_administration_legitimacy > 80
						var:roman_administration_legitimacy <= 90
					}					
				}
				add = 250
			}
			if = {
				limit = {
					scope:candidate = {
						has_variable = roman_administration_legitimacy
						var:roman_administration_legitimacy > 90
						var:roman_administration_legitimacy <= 100
					}					
				}
				add = 300
			}
			if = {
				limit = {
					scope:candidate = {
						has_variable = roman_administration_legitimacy
						var:roman_administration_legitimacy > 100
						var:roman_administration_legitimacy <= 110
					}					
				}
				add = 350
			}
			if = {
				limit = {
					scope:candidate = {
						has_variable = roman_administration_legitimacy
						var:roman_administration_legitimacy > 110
						var:roman_administration_legitimacy <= 120
					}					
				}
				add = 400
			}
			if = {
				limit = {
					scope:candidate = {
						has_variable = roman_administration_legitimacy
						var:roman_administration_legitimacy > 120
						var:roman_administration_legitimacy <= 130
					}					
				}
				add = 450
			}
			if = {
				limit = {
					scope:candidate = {
						has_variable = roman_administration_legitimacy
						var:roman_administration_legitimacy > 130
						var:roman_administration_legitimacy <= 140
					}					
				}
				add = 500
			}
			if = {
				limit = {
					scope:candidate = {
						has_variable = roman_administration_legitimacy
						var:roman_administration_legitimacy > 140
						var:roman_administration_legitimacy <= 150
					}					
				}
				add = 550
			}
			if = {
				limit = {
					scope:candidate = {
						has_variable = roman_administration_legitimacy
						var:roman_administration_legitimacy >= 150
					}					
				}
				add = 600
			}
		}
	}

	## Physical properties
	modifier = {
		add = -100
		desc = tooltip_imperial_vote_strength_emperor_deformed		
		NOT = { this = scope:candidate }
		
		scope:candidate = {
			OR = {
				has_trait = infirm
				has_trait = disfigured
				has_trait = blind
				has_trait = eunuch
				has_trait = dwarf
				has_trait = maimed
				has_trait = wounded_3
				has_trait = clubfooted
				has_trait = lisping
				has_trait = inbred
				has_trait = hunchbacked
				has_trait = scaly
				has_trait = albino
				has_trait = bleeder
				has_trait = gout_ridden
			}
		}
	}

	modifier = {
		add = 50
		desc = tooltip_imperial_vote_strength_comely

		scope:candidate = {
			OR = {
				has_trait = beauty_good_1
				has_trait = beauty_good_2
				has_trait = beauty_good_3
			}
		}
	}

	## Porphyrogénnētos
	modifier = {
		add = 50
		desc = tooltip_imperial_vote_strength_candidate_purple
		#NOT = { this = scope:candidate }

		scope:candidate = {
			has_trait = born_in_the_purple
		}
	}

	## Martial qualities
	modifier = {	# Candidate's martial
		desc = tooltip_imperial_vote_strength_commander_martial
		NOT = { this = scope:candidate } #Not on yourself.
		scope:candidate = { martial >= 10 }
		add = {
			value = 5
			if = {
				limit = {
					scope:candidate = { martial > 12 }
				}
				add = 6
			}
			if = {
				limit = {
					scope:candidate = { martial > 16 }
				}
				add = 8
			}
			if = {
				limit = {
					scope:candidate = { martial > 18 }
				}
				add = 9
			}
			if = {
				limit = {
					scope:candidate = { martial > 20 }
				}
				add = 10
			}
			if = {
				limit = {
					scope:candidate = { martial > 25 }
				}
				add = 13
			}
			if = {
				limit = {
					scope:candidate = { martial > 30 }
				}
				add = 15
			}
		}
	}
}
