	every_held_title = {
		if = {
			limit = {
				tier = tier_county
			}
			title_province = {
				add_to_global_variable_list = {
					name = domain_province_list
					target = this
				}					
			}
		}
	}
	travel_distance_dijkstra_white = {
		white_type = variable
		white_name = domain_province_list
		start = root.capital_province
		max = 10000
		return_dist = am_comm_malus_to_capital
		return_prev = bar
	}
	every_held_title = {
		if = {
			limit = {
				tier = tier_county
			}
			title_province = {
				if = {
					limit = {
						has_variable = am_comm_malus_to_capital
					}
					prev = {
						# set_global_variable = {
